def self.persist(score_card)
  ->(placement, value) {
    old_values = score_card.to_hash
    new_values = { placement => value }
    score_card = new(old_values.merge(new_values))
  }
end

def to_hash
  Yahtzee::SCORINGS.reduce({}) do |hash, attr|
    hash.merge(attr => send(attr))
  end
end
